<script type="text/babel">

const { useState, useEffect, useRef, useMemo } = React;
const { X, Save, Upload, Plus, Trash2, Film, Tv, Settings, Image: ImageIcon, Loader2, Download, UploadCloud, Search } = window.Lucide;

// ===== Mock Data & Types =====
const MOVIES = [
  { id: '1', type: 'movie', title: 'Sample Movie', description: 'A sample movie', genre: ['Action'], duration: '2h 10m', rating: 4.5, year: 2023, imageUrl: '', videoUrl: '', cast: ['Actor 1'], director: 'Director 1', episodes: [] }
];

function App() {
  const [movies, setMovies] = useState(() => {
    const saved = localStorage.getItem('streamgenius_content');
    return saved ? JSON.parse(saved) : MOVIES;
  });
  const [siteConfig, setSiteConfig] = useState(() => {
    const saved = localStorage.getItem('streamgenius_config');
    return saved ? JSON.parse(saved) : { siteName: 'SeriesGenius', siteLogoUrl: '' };
  });
  const [isAdmin, setIsAdmin] = useState(false);
  const [selectedMovie, setSelectedMovie] = useState(null);
  const [selectedEpisodeId, setSelectedEpisodeId] = useState(undefined);
  const [isAdminOpen, setIsAdminOpen] = useState(false);
  const [isPinModalOpen, setIsPinModalOpen] = useState(false);
  const [editingMovie, setEditingMovie] = useState(null);
  const [searchQuery, setSearchQuery] = useState('');

  useEffect(() => { localStorage.setItem('streamgenius_content', JSON.stringify(movies)); }, [movies]);
  useEffect(() => { localStorage.setItem('streamgenius_config', JSON.stringify(siteConfig)); document.title = siteConfig.siteName; }, [siteConfig]);

  const handleMovieClick = (movie) => { setSelectedMovie(movie); setSelectedEpisodeId(undefined); };
  const handleCloseModal = () => { setSelectedMovie(null); setSelectedEpisodeId(undefined); };

  const handleOpenAdmin = (movieToEdit) => { setEditingMovie(movieToEdit || null); setIsAdminOpen(true); };
  const handleSaveMovie = (movie) => {
    if (editingMovie) setMovies(prev => prev.map(m => m.id === movie.id ? movie : m));
    else setMovies(prev => [movie, ...prev]);
    setIsAdminOpen(false); setEditingMovie(null);
  };
  const handleSaveConfig = (config) => { setSiteConfig(config); setIsAdminOpen(false); };
  const handleAdminToggle = () => { setIsAdmin(!isAdmin); };

  const filteredMovies = useMemo(() => {
    if (!searchQuery) return movies;
    const q = searchQuery.toLowerCase();
    return movies.filter(m => m.title.toLowerCase().includes(q) || m.genre.some(g => g.toLowerCase().includes(q)));
  }, [movies, searchQuery]);

  const contentRows = useMemo(() => {
    if (searchQuery) return [{ title: `Search: "${searchQuery}"`, movies: filteredMovies }];
    if (movies.length === 0) return [];
    const series = movies.filter(m => m.type === 'series');
    const action = movies.filter(m => m.genre.includes('Action'));
    const recent = [...movies].reverse().slice(0,5);
    return [{title:'Trending', movies}, {title:'Series', movies: series}, {title:'Action', movies: action}, {title:'Recently Added', movies: recent}].filter(r=>r.movies.length>0);
  }, [movies, filteredMovies, searchQuery]);

  const featuredMovie = movies[0] || null;

  return (
    <div className="min-h-screen pb-20">
      {/* Navbar */}
      <div className="flex justify-between p-4 bg-gray-900 sticky top-0 z-20">
        <div className="flex items-center gap-2">
          {siteConfig.siteLogoUrl && <img src={siteConfig.siteLogoUrl} className="h-8 w-auto" />}
          <span className="font-bold text-xl">{siteConfig.siteName}</span>
        </div>
        <div className="flex gap-2">
          <input placeholder="Search..." className="px-2 rounded bg-gray-800" value={searchQuery} onChange={e=>setSearchQuery(e.target.value)} />
          <button onClick={()=>handleAdminToggle()} className="bg-indigo-600 px-3 py-1 rounded">Admin</button>
        </div>
      </div>

      {/* Hero */}
      {!searchQuery && featuredMovie && (
        <div className="p-4 bg-gray-800 m-4 rounded">
          <h2 className="text-xl font-bold">{featuredMovie.title}</h2>
          <p>{featuredMovie.description}</p>
          <button onClick={()=>handleMovieClick(featuredMovie)} className="mt-2 bg-indigo-600 px-3 py-1 rounded">Play</button>
        </div>
      )}

      {/* Rows */}
      <div className="space-y-6 mt-4">
        {contentRows.map((row, idx)=>(
          <div key={idx} className="px-4">
            <h3 className="font-bold mb-2">{row.title}</h3>
            <div className="flex gap-2 overflow-x-auto">
              {row.movies.map(m=>(
                <div key={m.id} className="flex-none w-32 bg-gray-700 p-2 rounded cursor-pointer" onClick={()=>handleMovieClick(m)}>
                  <div className="h-40 bg-gray-600 mb-2">{/* Poster placeholder */}</div>
                  <span>{m.title}</span>
                </div>
              ))}
            </div>
          </div>
        ))}
      </div>

      {/* Footer */}
      <footer className="mt-8 text-center text-gray-500 text-sm border-t border-gray-700 py-4">
        &copy; {new Date().getFullYear()} {siteConfig.siteName}. All rights reserved.
      </footer>

      {/* Modals */}
      {selectedMovie && <MovieDetailModal movie={selectedMovie} onClose={handleCloseModal} />}
      {isAdminOpen && <AdminModal movie={editingMovie} allMovies={movies} currentSiteConfig={siteConfig} onSaveContent={handleSaveMovie} onSaveConfig={handleSaveConfig} onClose={()=>setIsAdminOpen(false)} />}
    </div>
  );
}

// ===== AdminModal Simplified =====
function AdminModal({ movie, allMovies=[], currentSiteConfig, onSaveContent, onSaveConfig, onClose }) {
  const [activeTab, setActiveTab] = useState('content');
  const [formData, setFormData] = useState({...movie} || { title:'', genre:[], rating:0, episodes:[] });
  const [contentType, setContentType] = useState(movie?.type || 'movie');
  const fileInputRef = useRef(null);

  const handleChange = (e) => { const { name, value } = e.target; setFormData(prev=>({...prev,[name]:value})); };
  const handleSave = () => { onSaveContent({...formData,type:contentType,id:movie?.id || Date.now().toString()}); };

  return (
    <div className="fixed inset-0 bg-black/80 flex items-center justify-center z-50 p-4">
      <div className="bg-gray-900 p-4 rounded-lg w-full max-w-md max-h-[90vh] overflow-y-auto">
        <div className="flex justify-between mb-4">
          <h2 className="font-bold">{movie?'Edit':'Add'} Content</h2>
          <button onClick={onClose}><X className="w-6 h-6" /></button>
        </div>
        <div className="flex gap-2 mb-4">
          <button onClick={()=>setContentType('movie')} className={`px-2 py-1 rounded ${contentType==='movie'?'bg-indigo-600':'bg-gray-700'}`}>Movie</button>
          <button onClick={()=>setContentType('series')} className={`px-2 py-1 rounded ${contentType==='series'?'bg-indigo-600':'bg-gray-700'}`}>Series</button>
        </div>
        <div className="space-y-2">
          <input placeholder="Title" name="title" value={formData.title} onChange={handleChange} className="w-full p-2 rounded bg-gray-800" />
          <input placeholder="Director" name="director" value={formData.director || ''} onChange={handleChange} className="w-full p-2 rounded bg-gray-800" />
          <textarea placeholder="Description" name="description" value={formData.description || ''} onChange={handleChange} className="w-full p-2 rounded bg-gray-800" rows="3"></textarea>
          <button onClick={handleSave} className="bg-green-600 px-3 py-1 rounded mt-2">Save</button>
        </div>
      </div>
    </div>
  );
}

// ===== MovieDetailModal Simplified =====
function MovieDetailModal({ movie, onClose }) {
  return (
    <div className="fixed inset-0 bg-black/70 flex items-center justify-center z-40 p-4">
      <div className="bg-gray-800 p-4 rounded w-full max-w-sm">
        <div className="flex justify-between mb-2">
          <h2 className="font-bold">{movie.title}</h2>
          <button onClick={onClose}><X className="w-6 h-6" /></button>
        </div>
        <p>{movie.description}</p>
      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById('root')).render(<App />);
</script>
